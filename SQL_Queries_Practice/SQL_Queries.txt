1. Find Second Highest Salary
SELECT MAX(Salary) AS SecondHighestSalary
FROM Employee
WHERE Salary < (
    SELECT MAX(Salary)
    FROM Employee
);

2. Find Duplicate Records
SELECT Name,
       COUNT(*) AS DuplicateCount
FROM Employee
GROUP BY Name
HAVING COUNT(*) > 1;

3. Find Employees With Highest Salary
SELECT *
FROM Employee
WHERE Salary = (
    SELECT MAX(Salary)
    FROM Employee
);

4. Employees Whose Name Starts With 'A'
SELECT *
FROM Employee
WHERE Name LIKE 'A%';

5. Get Top 3 Salaries
SELECT DISTINCT TOP 3 Salary
FROM Employee
ORDER BY Salary DESC;

6. Find Total Salary Department Wise
SELECT Department,
       SUM(Salary) AS TotalSalary
FROM Employee
GROUP BY Department;

7. Maximum Salary Department Wise
SELECT Department,
       MAX(Salary) AS HighestSalary
FROM Employee
GROUP BY Department;

8. Find Average Salary
SELECT AVG(Salary) AS AverageSalary
FROM Employee;

9. Employees Earning More Than Average Salary
SELECT *
FROM Employee
WHERE Salary > (
    SELECT AVG(Salary)
    FROM Employee
);

10. Update Salary By 10%
UPDATE Employee
SET Salary = Salary + (Salary * 0.10);
